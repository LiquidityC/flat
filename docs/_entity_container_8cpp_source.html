425 void
426 EntityContainer::iterateCollidablesFor(const Entity* source, EntityIter func)
427 {
428  const EntityProperties::Areas& currentAreas =
429  source->getEntityProperties().getCurrentAreas();
430  std::map<float, Entity*> sortedMap;
431  EntityShape colliderShape =
432  source->getEntityProperties().getColliderShape();
433  float sx = static_cast<float>(colliderShape.x + (colliderShape.w / 2));
434  float sy = static_cast<float>(colliderShape.y + (colliderShape.h / 2));
435 
436  // Itterate the objects and sort them according to distance
437  for (auto& area : currentAreas) {
438  for (auto& object : spatialPartitionMap[area]) {
439  if (!object.second->getEntityProperties().isCollidable()) {
440  continue;
441  }
442  if (*source == *object.second) {
443  continue;
444  }
445 
446  const EntityShape& targetShape =
447  object.second->getEntityProperties().getColliderShape();
448  float tx = static_cast<float>(targetShape.x + (targetShape.w / 2));
449  float ty = static_cast<float>(targetShape.y + (targetShape.h / 2));
450 
451  float distance;
452  if (sx == tx) {
453  distance = static_cast<float>(std::abs(sy - ty));
454  } else if (sy == ty) {
455  distance = static_cast<float>(std::abs(sx - tx));
456  } else {
457  distance = sqrt(pow(sx - tx, 2) + pow(sy - ty, 2));
458  }
459 
460  if (sortedMap.find(distance) != sortedMap.end()) {
461  if (*sortedMap[distance] == *object.second) {
462  continue;
463  }
464  }
465 
466  while (sortedMap.find(distance) != sortedMap.end()) {
467  distance += 0.00001f;
468  }
469 
470  sortedMap[distance] = object.second;
471  }
472  }
473 
474  // Itterate the sorted objects and call the cb function
475  for (auto objectIter = sortedMap.begin(); objectIter != sortedMap.end();
476  objectIter++) {
477  func(objectIter->second);
478  }
479 }
